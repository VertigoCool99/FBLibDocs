Introduction

First Require The Library And Make A Window And Tabs/Groupboxes


local Library = "Library Loadstring Here"
local Window = Library:CreateWindow({Title=String,TweenTime=Int,Center=Bool})
                        
local NewTab = Window2:AddTab(String(Name))
                        
local NewGroupbox = NewTab:AddLeftGroupbox(String(Name)) --Left Side
Local NewGroupbox2 = NewTab:AddRightGroupbox(String(Name)) --Right Side

Label


local NewLabel = NewGroupbox:AddLabel(String(Index),String(Text))

NewLabel:SetText(String(Text)) --Updates The Label

Labels Can Take Addons Such As KeyPickers And ColorPickers!

Button


local NewButton = NewGroupbox:AddButton({
    Text=String,
    Func=Function,
})

NewButton:AddButton({ --Add A Sub Button
    Text = String,
    Func= Function,
})

NewButton:AddTooltip(String)

Toggle


local NewToggle = NewGroupbox:AddToggle(String(Index),{
    Text = String,
    Default = Bool,
    Risky = Bool,
    Callback = Function,
})

NewToggle:SetValue(Bool)
NewToggle:AddTooltip(String)
