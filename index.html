Introduction

First Require The Library And Make A Window And Tabs/Groupboxes


local Library = "Library Loadstring Here"
local Window = Library:CreateWindow({Title=String,TweenTime=Int,Center=Bool})
                        
local NewTab = Window2:AddTab(String(Name))
                        
local NewGroupbox = NewTab:AddLeftGroupbox(String(Name)) --Left Side
local NewGroupbox2 = NewTab:AddRightGroupbox(String(Name)) --Right Side

Divider


NewGroupbox:AddDivider()

Label


local NewLabel = NewGroupbox:AddLabel(String(Index),String(Text))

NewLabel:SetText(String(Text)) --Updates The Label

Labels Can Take Addons Such As KeyPickers And ColorPickers!

Button


local NewButton = NewGroupbox:AddButton({
    Text = String,
    Func = Function,
})

NewButton:AddButton({ --Add A Sub Button
    Text = String,
    Func = Function,
})

NewButton:AddTooltip(String)

Toggle


local NewToggle = NewGroupbox:AddToggle(String(Index),{
    Text = String,
    Default = Bool,
    Risky = Bool,
    Callback = Function,
})

NewToggle:SetValue(Bool)
NewToggle:AddTooltip(String)
NewToggle:OnChanged(function(Bool)
    print(Bool)
end)

Toggles Can Take Addons Such As KeyPickers And ColorPickers!

Slider


local NewSlider = NewGroupbox:AddSlider(String(Index),{
    Text = String,
    Default = Int,
    Min = Int,
    Max = Int,
    Rounding = Int,
    Callback = Function,
})

NewSlider:AddToolTip(String)
NewSlider:SetValue(Int)
NewSlider:OnChanged(function(Int)
    print(Int) 
end)

Dropdown


local NewDropdown = NewGroupbox:AddDropdown(String(Index),{
    Text = String;
    SpecialType = "Players" or "Team" --Not Needed
    AllowNull = Bool,
    Values = Table,
    Multi = Bool,
    Callback = Function,
})

NewDropdown:SetValues(Table) --Sets The List
NewDropdown:SetValue(String) --Sets Current Selected
NewDropdown:OnChanged(function(Value)
    print(Value)
end)

Textbox


local NewTextbox = NewGroupbox:AddInput(String(Index),{
    Text = String;
    Default = String,
    Numeric = Bool, --Not Needed
    Finished = Bool, --Not Needed
    Callback = Function,
})

NewTextbox:AddTooltip(String)
NewTextbox:OnChanged(function(Value)
    print(Value)
end)

Color Picker


local NewColorPicker = NewGroupbox:AddColorPicker(String(Index),{
    Text = String;
    Default = Color3;
    Rainbow = Bool, --Not Needed
    Title = String, --Not Needed
    Callback = Function,
})

ColorPicker:SetValueRGB(Color3RGB)
ColorPicker:OnChanged(function(Value)
    print(Value)    
end)

Key Picker [ADDON]


local ToggleKeyPicker = NewToggle:AddKeyPicker(String(Index),{
    Default = Key,
    Mode = Always | Toggle | Hold,
    IgnoreKeybindFrame = Bool --Not Needed [Wether To Show In KeyBind Frame]
    SyncToggleState = Bool, --Not Needed
    Callback = Function,
})

ToggleKeyPicker:SetValue({Key,Mode})
ToggleKeyPicker:OnChanged(function(Value)
    print(Value)
end)
