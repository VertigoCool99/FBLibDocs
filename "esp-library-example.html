Introduction

First Require The Library And Make A Your Players Functtion


local EspLibrary, EspLibraryFunctions = "Library Loadstring Here"
local Players = cloneref(game:GetService("Players"))
  
local PlayerUpdater = game:GetService("RunService").RenderStepped
PlayerUpdater:Connect(EspLibrary.Esp.UpdateEsp)

for i,v in ipairs(Players:GetPlayers()) do
    if v ~= LocalPlayer and v.Character ~= nil then
        if v:IsFriendsWith(LocalPlayer.UserId) then
            table.insert(EspLibrary.Friends,v.Character)
        end
        EspLibrary.Esp:CreateEsp(v.Character) --Takes Model
    end
end

Players.PlayerAdded:Connect(function(plr)
    plr.CharacterAdded:Connect(function(char)
        if plr:IsFriendsWith(LocalPlayer.UserId) then
            table.insert(EspLibrary.Friends,char)
        end
        if not EspLibrary.PlayerCache[char] then
            EspLibrary.Esp:CreateEsp(char) --Takes Model
        end
    end)
end)
