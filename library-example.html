Example


local Library = loadstring(game:HttpGet(""))()

local Window = Library:CreateWindow({Title=" Example Ui",TweenTime=.15,Center=true})
   
local NewTab = Window:AddTab("Main")
                        
local NewGroupbox = NewTab:AddLeftGroupbox("First Group")

local NewLabel = NewGroupbox:AddLabel("NewLabel","TextLabel!")
local NewButton = NewGroupbox:AddButton({Text = "Button",Func = function()
    print("Clicked!")
end,})
NewButton:AddButton({Text = "Button 2",Func = function()
    print("Clicked")
end,})

NewGroupbox:AddDivider()

local NewToggle = NewGroupbox:AddToggle("NewToggle",{Text = "Toggle",Default = false,Risky = false})
NewToggle:AddTooltip("Hovering Text Above Toggle")
NewToggle:OnChanged(function(value)
    print(value)
end)

local NewSlider = NewGroupbox:AddSlider("NewSlider",{Text = "Slider",Default = 30,Min = 0,Max = 100,Rounding = 0})
NewSlider:OnChanged(function(Value)
    print(Value)
end)

local NewDropdown = NewGroupbox:AddDropdown("NewDropdown",{Text = "Dropdown", AllowNull = true,Values = {"a","b","c"},Multi = false,})
NewDropdown:OnChanged(function(Value)
    print(Value)
end)

local NewTextbox = NewGroupbox:AddInput("NewTextbox",{Text = "TextBox";Default = "Texting",Numeric = false,Finished = true})
NewTextbox:OnChanged(function(Value)
    print(Value)
end)

local NewColorPicker = NewGroupbox:AddColorPicker("NewColorPicker",{Text = "ColorPicker";Default = Color3.fromRGB(255,55,65);Rainbow = false,Title = "Picker Title"})
NewColorPicker:OnChanged(function(Color)
    print(Color)
end)

--Addons
local ToggleKeyPicker = NewToggle:AddKeyPicker("NewToggleKeyPicker",{Default = "K",Mode = "Toggle",SyncToggleState = true})
Library.Options["NewToggleKeyPicker"]:OnClick(function()
    print("Key Pressed")
end)

local NewToggleColorPicker = NewToggle:AddColorPicker("NewToggleColorPicker",{Default = Color3.fromRGB(255,55,65);Rainbow = false})
NewToggleColorPicker:OnChanged(function(Color)
    print(Color)
end)


Library:SetWatermark("Float.Balls Example Ui")

--Settings Start
local Settings = Window:AddTab("Settings")
local SettingsUI = Settings:AddLeftGroupbox("UI")

local SettingsUnloadButton = SettingsUI:AddButton({Text="Unload",Func=function()
    Library:Unload()
end})

local SettingsMenuLabel = SettingsUI:AddLabel("SettingsMenuKeybindLabel","Menu Keybind")
local SettingsMenuKeyPicker = SettingsMenuLabel:AddKeyPicker("SettingsMenuKeyBind",{Default="Insert",IgnoreKeybindFrame=true})
Library.Options["SettingsMenuKeyBind"]:OnClick(function()
    Library:Toggle()
end)
local SettingsNotiPositionDropdown = SettingsUI:AddDropdown("SettingsNotiPositionDropdown",{Text="Notification Position",Values={"Top_Left","Top_Right","Bottom_Left","Bottom_Right"},Default="Top_Left"})
SettingsNotiPositionDropdown:OnChanged(function(Value)
    Library.NotificationPosition = Value
end)

Library.ThemeManager:SetLibrary(Library)
Library.SaveManager:SetLibrary(Library)
Library.ThemeManager:ApplyToTab(Settings)
Library.SaveManager:IgnoreThemeSettings()
Library.SaveManager:SetIgnoreIndexes({"MenuKeybind","BackgroundColor", "ActiveColor", "ItemBorderColor", "ItemBackgroundColor", "TextColor" , "DisabledTextColor", "RiskyColor"})
Library.SaveManager:SetFolder('Test')
Library.SaveManager:BuildConfigSection(Settings)
--Settings End
